BYM<-nimbleCode({
for(i in 1: n){
obs[i]~dpois(mu[i])
log(mu[i])<-log(expe[i])+log(theta[i]) 
theta[i]<-exp(ac0+bym[i]) 
vc[i]~dnorm(0,tauV)
bym[i]<-vc[i]+uc[i]
}  
for (k in 1 :nsumN){wei[k]<-1} 
uc[1:n]~dcar_normal(adj[1:nsumN],wei[1:nsumN],num[1:n],tauU,zero_mean=1)  
tauU<-pow(sdU,-2)
tauV<-pow(sdV,-2)
sdV~dunif(0,10)
sdU~dunif(0,10)
ac0~dnorm(0,tau0)
sd0~dunif(0,10)
})
